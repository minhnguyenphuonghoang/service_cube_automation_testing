Add cookies to browser
    [Arguments]    ${cookies}
    @{list_of_cookies}=    Split String    ${cookies}    ;
    :FOR    ${a_cookie}    IN    @{list_of_cookies}
    \    ${cookie_name}    ${cookie_value}    Split String    ${a_cookie.strip()}    =
    \    Add Cookie    ${cookie_name}    ${cookie_value}
